//Task 19
public class task19 {
public static void main(String[] args) {
  int a = 8;
  int b = 3;
  int c = a/2;
  double one_hexagon_side = Math.sqrt((b*b)+(c*c)); //Using pythagoras formula to get one side of the hexagon
  double area_of_hexagon = (3*Math.sqrt(3)*(one_hexagon_side*one_hexagon_side))/2;
  double circumference_of_hexagon = 6*one_hexagon_side;
 System.out.println("One Hexagon Side = "+one_hexagon_side);
 System.out.println("Area Of Hexagon = "+area_of_hexagon);
 System.out.println("Circumference Of Hexagon = "+circumference_of_hexagon);
  }
}
